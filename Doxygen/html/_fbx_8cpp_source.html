53 void Fbx::processNode(aiNode* node, const aiScene* scene, glm::mat4 transform)
54 {
55  // process each mesh located at the current node
56  for (unsigned int i = 0; i < node->mNumMeshes; i++)
57  {
58  /// the node object only contains indices to index the actual objects in the scene.
59  /// the scene contains all the data, node is just to keep stuff organized (like relations between nodes).
60  aiMesh* mesh = scene->mMeshes[node->mMeshes[i]];
61  meshes.push_back(processMesh(mesh, scene, transform * to_glm(node->mTransformation)));
62  }
63  /// after we've processed all of the meshes (if any) we then recursively process each of the children nodes
64  for (unsigned int i = 0; i < node->mNumChildren; i++)
65  {
66  processNode(node->mChildren[i], scene, transform * to_glm(node->mTransformation));
67  }
68 }

71 Mesh Fbx::processMesh(aiMesh* mesh, const aiScene* scene, glm::mat4 transform)
72 {
73  std::vector<Vertex> vertices;
74  std::vector<unsigned int> indices;
75  std::vector<TextureMesh> textures;
76 
77  /// walk through each of the mesh's vertices
78  for (unsigned int i = 0; i < mesh->mNumVertices; i++)
79  {
80  Vertex vertex;
81 
82  /// we declare a placeholder vector since assimp uses its own vector class that doesn't directly convert to glm's vec3 class so we transfer the data to this placeholder glm::vec3 first.
83  glm::vec3 vector;
84 
85  /// positions
86  if (mesh->HasPositions())
87  {
88  vector.x = mesh->mVertices[i].x;
89  vector.y = mesh->mVertices[i].y;
90  vector.z = mesh->mVertices[i].z;
91  vertex.Position = vector;
92  }
93 
94  /// normals
95  if (mesh->HasNormals()) {
96  vector.x = mesh->mNormals[i].x;
97  vector.y = mesh->mNormals[i].y;
98  vector.z = mesh->mNormals[i].z;
99  vertex.Normal = vector;
100  }
101  /// texture coordinates
102  if (mesh->mTextureCoords[0])
103  {
104  glm::vec2 vec;
105  /// a vertex can contain up to 8 different texture coordinates. We thus make the assumption that we won't
106  /// use models where a vertex can have multiple texture coordinates so we always take the first set (0).
107  vec.x = mesh->mTextureCoords[0][i].x;
108  vec.y = mesh->mTextureCoords[0][i].y;
109  vertex.TexCoords = vec;
110  }
111  else
112  vertex.TexCoords = glm::vec2(0.0f, 0.0f);
113 
114  if (mesh->HasTangentsAndBitangents())
115  {
116  /// tangent
117  vector.x = mesh->mTangents[i].x;
118  vector.y = mesh->mTangents[i].y;
119  vector.z = mesh->mTangents[i].z;
120  vertex.Tangent = vector;
121  /// bitangent
122  vector.x = mesh->mBitangents[i].x;
123  vector.y = mesh->mBitangents[i].y;
124  vector.z = mesh->mBitangents[i].z;
125  vertex.Bitangent = vector;
126  }
127  vertices.push_back(vertex);
128  }
129 
130 
131  /// now run through each of the mesh's faces (a face is a mesh its triangle) and retrieve the corresponding vertex indices.
132  for (unsigned int i = 0; i < mesh->mNumFaces; i++)
133  {
134  aiFace face = mesh->mFaces[i];
135  /// retrieve all indices of the face and store them in the indices vector
136  for (unsigned int j = 0; j < face.mNumIndices; j++)
137  {
138  indices.push_back(face.mIndices[j]);
139  }
140  }
141 
142  /// process materials
143  aiMaterial* material = scene->mMaterials[mesh->mMaterialIndex];
144 
145 
146  /// 1. diffuse maps
147  std::vector<TextureMesh> diffuseMaps = loadMaterialTextures(material, aiTextureType_DIFFUSE, "texture_diffuse");
148  textures.insert(textures.end(), diffuseMaps.begin(), diffuseMaps.end());
149  /// 2. specular maps
150  std::vector<TextureMesh> specularMaps = loadMaterialTextures(material, aiTextureType_SPECULAR, "texture_specular");
151  textures.insert(textures.end(), specularMaps.begin(), specularMaps.end());
152  /// 3. normal maps
153  std::vector<TextureMesh> normalMaps = loadMaterialTextures(material, aiTextureType_HEIGHT, "texture_normal");
154  textures.insert(textures.end(), normalMaps.begin(), normalMaps.end());
155  /// 4. height maps
156  std::vector<TextureMesh> heightMaps = loadMaterialTextures(material, aiTextureType_AMBIENT, "texture_height");
157  textures.insert(textures.end(), heightMaps.begin(), heightMaps.end());
158  Mesh temp(vertices, indices, textures, transform);
159  render->SetupMesh(temp.VAO, temp.VBO, temp.EBO, temp.vertices, temp.indices);
160 
161  /// return a mesh object created from the extracted mesh data
162  return temp;
163 }

167 std::vector<TextureMesh> Fbx::loadMaterialTextures(aiMaterial* mat, aiTextureType type, std::string typeName)
168 {
169  std::vector<TextureMesh> textures;
170  int count = mat->GetTextureCount(type);
171  if (Debugger::GetInstance()->debugToConsole) std::cout << "Model.cpp::INFO:: " << typeName << " " << count << std::endl;
172 
173  for (unsigned int i = 0; i < mat->GetTextureCount(type); i++)
174  {
175  aiString str;
176  mat->GetTexture(type, i, &str);
177  /// check if texture was loaded before and if so, continue to next iteration: skip loading a new texture
178  bool skip = false;
179  for (unsigned int j = 0; j < textures_loaded.size(); j++)
180  {
181  if (std::strcmp(textures_loaded[j].path.data(), str.C_Str()) == 0)
182  {
183  textures.push_back(textures_loaded[j]);
184  skip = true; /// a texture with the same filepath has already been loaded, continue to next one. (optimization)
185  break;
186  }
187  }
188  if (!skip)
189  { /// if texture hasn't been loaded already, load it
190 
191  TextureMesh texture;
192  texture.id = render->TextureFromFile(str.C_Str(), this->directory);
193 
194  texture.type = typeName;
195  texture.path = str.C_Str();
196  textures.push_back(texture);
197  textures_loaded.push_back(texture); /// store it as texture loaded for entire model, to ensure we won't unnecesery load duplicate textures.
198  }
199  }
200  return textures;
201 }
