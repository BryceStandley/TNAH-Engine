333 // Simple helper function to load an image into a OpenGL texture with common settings
334 bool MainMenuGUI::LoadTextureFromFile(std::string filename, GLuint* out_texture, int* out_width, int* out_height)
335 {
336  if (filename.find(".png") != std::string::npos) // if the image is a png, flip
337  {
338  stbi_set_flip_vertically_on_load(false);
339  }
340 
341  // Load from file
342  int image_width = 0;
343  int image_height = 0;
344  unsigned char* image_data = stbi_load(filename.c_str() , &image_width, &image_height, nullptr, 4);
345  if (image_data == nullptr)
346  return false;
347 
348  // Create a OpenGL texture identifier
349  GLuint image_texture;
350  glGenTextures(1, &image_texture);
351  glBindTexture(GL_TEXTURE_2D, image_texture);
352 
353  // Setup filtering parameters for display
354  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
355  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
356  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE); // This is required on WebGL for non power-of-two textures
357  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE); // Same
358 
359  // Upload pixels into texture
360 #if defined(GL_UNPACK_ROW_LENGTH) && !defined(__EMSCRIPTEN__)
361  glPixelStorei(GL_UNPACK_ROW_LENGTH, 0);
362 #endif
363  glTexImage2D(GL_TEXTURE_2D, 0, GL_RGBA, image_width, image_height, 0, GL_RGBA, GL_UNSIGNED_BYTE, image_data);
364  stbi_image_free(image_data);
365 
366  *out_texture = image_texture;
367  *out_width = image_width;
368  *out_height = image_height;
369 
370  return true;
371 }
