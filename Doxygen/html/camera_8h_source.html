113  if (direction == FORWARD)

168  /**
169  * @brief calculates and updates vectors
170  */
171  void updateCameraVectors()
172  {
173  /// calculate the new Front vector
174  glm::vec3 front;
175  front.x = cos(glm::radians(Yaw)) * cos(glm::radians(Pitch));
176  front.y = sin(glm::radians(Pitch));
177  front.z = sin(glm::radians(Yaw)) * cos(glm::radians(Pitch));
178  Front = glm::normalize(front);
179  /// also re-calculate the Right and Up vector
180  /// normalize the vectors, because their length gets closer to 0 the more you look up or down which results in slower movement.
181  Right = glm::normalize(glm::cross(Front, WorldUp));
182  Up = glm::normalize(glm::cross(Right, Front));
