349 float Scene::BilinearInterpolation(glm::vec3 p)
350 {
351  int intX = p.x;
352  int intZ = p.z;
353  float fracX = p.x - intX;
354  float fracZ = p.z - intZ;
355 
356  float v1 = PlayerToTerrainPosition(glm::vec3(intX, 0, intZ), glm::vec3(0,0,0)).y;
357  float v2 = PlayerToTerrainPosition(glm::vec3(intX, 0, intZ), glm::vec3(1,0,0)).y;
358  float v3 = PlayerToTerrainPosition(glm::vec3(intX, 0, intZ + 1), glm::vec3(0,0,1)).y;
359  float v4 = PlayerToTerrainPosition(glm::vec3(intX, 0, intZ), glm::vec3(1,0,1)).y;
360  float i1 = Interpolate(v1, v2, fracX);
361  float i2 = Interpolate(v3, v4, fracX);
362  float newHeight = Interpolate(i1, i2, fracZ);
363  return newHeight;

694 bool Scene::intersectRaySegmentSphere(glm::vec3 o, glm::vec3 d, glm::vec3 so, float radius, glm::vec3 &ip)
695 {
696  //we pass in d non-normalized to keep it's length
697  //then we use that length later to compare the intersection point to make sure
698  //we're within the actual ray segment
699  auto l = (float)sqrt(pow(d.x,2) + pow(d.y, 2) + pow(d.z, 2));
700  d /= l;
701 
702  glm::vec3 m = o - so;
703  float b = glm::dot(m, d);
704  float c = glm::dot(m, m) - radius;
705 
706  // Exit if r’s origin outside s (c > 0) and r pointing away from s (b > 0)
707  if(c > 0.0f && b > 0.0f)
708  return false;
709  float discr = b*b - c;
710 
711  // A negative discriminant corresponds to ray missing sphere
712  if(discr < 0.0f)
713  return false;
714 
715  // Ray now found to intersect sphere, compute smallest t value of intersection
716  float t = -b - sqrtf(discr);
717 
718  // If t is negative, ray started inside sphere so clamp t to zero
719  if(t < 0.0f)
720  t = 0.0f;
721  ip = o + (d * t);
722 
723  //here's that last segment check I was talking about
724  if(t > l)
725  return false;
726 
727  return true;
728 }
